

//Insert an element in an array at a given position

\#include <stdio.h>



int main() {

&nbsp;   int arr\[100]; // Declare an array with a maximum capacity of 100

&nbsp;   int n, element, pos, i;



&nbsp;   printf("Enter the number of elements in the array: ");

&nbsp;   scanf("%d", \&n);



&nbsp;   printf("Enter the elements of the array:\\n");

&nbsp;   for (i = 0; i < n; i++) {

&nbsp;       scanf("%d", \&arr\[i]);

&nbsp;   }



&nbsp;   printf("Enter the element to be inserted: ");

&nbsp;   scanf("%d", \&element);



&nbsp;   printf("Enter the position (0 to %d) to insert the element: ", n);

&nbsp;   scanf("%d", \&pos);



&nbsp;   // Validate the position

&nbsp;   if (pos < 0 || pos > n) {

&nbsp;       printf("Invalid position!\\n");

&nbsp;   } else {

&nbsp;       // Shift elements to the right to make space for the new element

&nbsp;       for (i = n; i > pos; i--) {

&nbsp;           arr\[i] = arr\[i - 1];

&nbsp;       }



&nbsp;       // Insert the new element

&nbsp;       arr\[pos] = element;

&nbsp;       n++; // Increment the number of elements



&nbsp;       printf("Array after insertion:\\n");

&nbsp;       for (i = 0; i < n; i++) {

&nbsp;           printf("%d ", arr\[i]);

&nbsp;       }

&nbsp;       printf("\\n");

&nbsp;   }



&nbsp;   return 0;

}
