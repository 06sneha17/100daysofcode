//Write a program to swap the first and last digit of a number

\#include <stdio.h>



int main() {

&nbsp;   int num, temp, firstDigit, lastDigit, digitsCount = 0;

&nbsp;   

&nbsp;   // Input number

&nbsp;   printf("Enter a number: ");

&nbsp;   scanf("%d", \&num);

&nbsp;   

&nbsp;   temp = num;

&nbsp;   

&nbsp;   // Count number of digits

&nbsp;   while (temp != 0) {

&nbsp;       temp /= 10;

&nbsp;       digitsCount++;

&nbsp;   }

&nbsp;   

&nbsp;   // Finds first and last digits of the input number

&nbsp;   firstDigit = num / (int)pow(10, digitsCount - 1);

&nbsp;   lastDigit = num % 10;

&nbsp;   

&nbsp;   // Swap first and last digits of the number

&nbsp;   temp = num - (firstDigit \* (int)pow(10, digitsCount - 1));

&nbsp;   temp += (lastDigit \* (int)pow(10, digitsCount - 1));

&nbsp;   temp -= lastDigit;

&nbsp;   temp += firstDigit;

&nbsp;   

&nbsp;   // Output the number with swapped digits

&nbsp;   printf("Number with first and last digits swapped: %d\\n", temp);

&nbsp;   

&nbsp;   return 0;

}
