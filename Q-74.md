//Find the transpose of a matrix

\#include <stdio.h>



\#define ROWS 3 // Define number of rows for the original matrix

\#define COLS 4 // Define number of columns for the original matrix



int main() {

&nbsp;   int originalMatrix\[ROWS]\[COLS] = {

&nbsp;       {1, 2, 3, 4},

&nbsp;       {5, 6, 7, 8},

&nbsp;       {9, 10, 11, 12}

&nbsp;   };



&nbsp;   // The transposed matrix will have dimensions COLS x ROWS

&nbsp;   int transposedMatrix\[COLS]\[ROWS];



&nbsp;   // Print the original matrix

&nbsp;   printf("Original Matrix:\\n");

&nbsp;   for (int i = 0; i < ROWS; i++) {

&nbsp;       for (int j = 0; j < COLS; j++) {

&nbsp;           printf("%d\\t", originalMatrix\[i]\[j]);

&nbsp;       }

&nbsp;       printf("\\n");

&nbsp;   }



&nbsp;   // Transpose the matrix

&nbsp;   for (int i = 0; i < ROWS; i++) {

&nbsp;       for (int j = 0; j < COLS; j++) {

&nbsp;           transposedMatrix\[j]\[i] = originalMatrix\[i]\[j];

&nbsp;       }

&nbsp;   }



&nbsp;   // Print the transposed matrix

&nbsp;   printf("\\nTransposed Matrix:\\n");

&nbsp;   for (int i = 0; i < COLS; i++) {

&nbsp;       for (int j = 0; j < ROWS; j++) {

&nbsp;           printf("%d\\t", transposedMatrix\[i]\[j]);

&nbsp;       }

&nbsp;       printf("\\n");

&nbsp;   }



&nbsp;   return 0;

}
