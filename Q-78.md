//Find the sum of main diagonal elements for a square matrix

\#include <stdio.h>



int main() {

&nbsp;   int matrix\[3]\[3] = { // Example 3x3 matrix

&nbsp;       {1, 2, 3},

&nbsp;       {4, 5, 6},

&nbsp;       {7, 8, 9}

&nbsp;   };

&nbsp;   int size = 3; // Size of the square matrix

&nbsp;   int mainDiagonalSum = 0;



&nbsp;   // Loop to iterate through the main diagonal elements

&nbsp;   for (int i = 0; i < size; i++) {

&nbsp;       // Add the element at matrix\[i]\[i] to the sum

&nbsp;       mainDiagonalSum += matrix\[i]\[i]; // \[1, 6]

&nbsp;   }



&nbsp;   // Print the result

&nbsp;   printf("Sum of Main Diagonal elements: %d\\n", mainDiagonalSum); // \[5]



&nbsp;   return 0;

}
