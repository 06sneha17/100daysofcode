

//Find the second largest element in an array

\#include <stdio.h>

\#include <limits.h> 



int findSecondLargest(int arr\[], int n) {

&nbsp;   if (n < 2) {

&nbsp;       printf("Array must have at least two elements.\\n");

&nbsp;       return -1; // Indicate an error or no second largest

&nbsp;   }



&nbsp;   int largest = INT\_MIN;

&nbsp;   int secondLargest = INT\_MIN;



&nbsp;   for (int i = 0; i < n; i++) {

&nbsp;       if (arr\[i] > largest) {

&nbsp;           secondLargest = largest; // The old largest becomes the second largest

&nbsp;           largest = arr\[i];        // The current element is the new largest

&nbsp;       } else if (arr\[i] > secondLargest \&\& arr\[i] != largest) {

&nbsp;           secondLargest = arr\[i]; // The current element is greater than secondLargest but not largest

&nbsp;       }

&nbsp;   }



&nbsp;   if (secondLargest == INT\_MIN) {

&nbsp;       printf("No distinct second largest element found (e.g., all elements are the same).\\n");

&nbsp;       return -1; // Indicate no distinct second largest

&nbsp;   }



&nbsp;   return secondLargest;

}



int main() {

&nbsp;   int arr1\[] = {12, 35, 1, 10, 34, 1};

&nbsp;   int n1 = sizeof(arr1) / sizeof(arr1\[0]);

&nbsp;   printf("Second largest in arr1: %d\\n", findSecondLargest(arr1, n1));



&nbsp;   int arr2\[] = {5, 5, 5, 5};

&nbsp;   int n2 = sizeof(arr2) / sizeof(arr2\[0]);

&nbsp;   printf("Second largest in arr2: %d\\n", findSecondLargest(arr2, n2));



&nbsp;   int arr3\[] = {10};

&nbsp;   int n3 = sizeof(arr3) / sizeof(arr3\[0]);

&nbsp;   printf("Second largest in arr3: %d\\n", findSecondLargest(arr3, n3));



&nbsp;   return 0;

}
