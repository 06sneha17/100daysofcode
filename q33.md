//Write a program to check if a number is an Armstrong number

\#include <stdio.h>

\#include <math.h> // Required for the pow() function



int main() {

&nbsp;   int number, originalNumber, remainder, result = 0, n = 0;



&nbsp;   // Input from the user

&nbsp;   printf("Enter a number: ");

&nbsp;   scanf("%d", \&number);



&nbsp;   originalNumber = number;



&nbsp;   // Count the number of digits

&nbsp;   // A temporary variable 'temp' can be used here to avoid modifying 'originalNumber' prematurely.

&nbsp;   int temp = number; 

&nbsp;   while (temp != 0) {

&nbsp;       temp /= 10;

&nbsp;       n++;

&nbsp;   }



&nbsp;   // Reset originalNumber for digit extraction and sum calculation

&nbsp;   temp = number; 



&nbsp;   // Calculate the sum of the nth powers of each digit

&nbsp;   while (temp != 0) {

&nbsp;       remainder = temp % 10;

&nbsp;       result += pow(remainder, n); // pow() returns a double, so careful with type casting if needed

&nbsp;       temp /= 10;

&nbsp;   }



&nbsp;   // Check if the number is an Armstrong number

&nbsp;   if (result == number) {

&nbsp;       printf("%d is an Armstrong number.\\n", number);

&nbsp;   } else {

&nbsp;       printf("%d is not an Armstrong number.\\n", number);

&nbsp;   }



&nbsp;   return 0;

}
