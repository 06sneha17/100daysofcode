//Count vowels and consonants in a string

\#include <stdio.h>

\#include <string.h> // Required for strlen()

\#include <ctype.h>  // Required for tolower()



int main() {

&nbsp;   char str\[100]; // Declare a character array to store the string

&nbsp;   int vowels = 0;

&nbsp;   int consonants = 0;

&nbsp;   int i;



&nbsp;   printf("Enter a string: ");

&nbsp;   // Use fgets for safer input compared to gets()

&nbsp;   // It prevents buffer overflow by limiting the input size

&nbsp;   fgets(str, sizeof(str), stdin);



&nbsp;   // Iterate through each character of the string

&nbsp;   // strlen(str) returns the length of the string

&nbsp;   for (i = 0; i < strlen(str); i++) {

&nbsp;       // Convert the character to lowercase for easier comparison

&nbsp;       char ch = tolower(str\[i]);



&nbsp;       // Check if the character is an alphabet

&nbsp;       if (ch >= 'a' \&\& ch <= 'z') {

&nbsp;           // Check if the character is a vowel

&nbsp;           if (ch == 'a' || ch == 'e' || ch == 'i' || ch == 'o' || ch == 'u') {

&nbsp;               vowels++;

&nbsp;           } else {

&nbsp;               // If it's an alphabet and not a vowel, it's a consonant

&nbsp;               consonants++;

&nbsp;           }

&nbsp;       }

&nbsp;       // Other characters (spaces, numbers, special characters) are ignored

&nbsp;   }



&nbsp;   printf("Number of Vowels: %d\\n", vowels);

&nbsp;   printf("Number of Consonants: %d\\n", consonants);



&nbsp;   return 0;

}
