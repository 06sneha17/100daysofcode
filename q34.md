//Write a program to check if a number is prime

\#include <stdio.h>

\#include <math.h> // Required for sqrt() function



int main() {

&nbsp;   int num, i, isPrime = 1;



&nbsp;   printf("Enter a positive integer: ");

&nbsp;   scanf("%d", \&num);



&nbsp;   // Handle special cases: 0, 1, and negative numbers

&nbsp;   if (num <= 1) {

&nbsp;       isPrime = 0; // Numbers less than or equal to 1 are not prime

&nbsp;   } else {

&nbsp;       // Check for divisibility from 2 up to the square root of the number

&nbsp;       for (i = 2; i \* i <= num; i++) {

&nbsp;           if (num % i == 0) {

&nbsp;               isPrime = 0; // If divisible, it's not prime

&nbsp;               break;       // Exit the loop as soon as a factor is found

&nbsp;           }

&nbsp;       }

&nbsp;   }



&nbsp;   // Print the result based on the isPrime flag

&nbsp;   if (isPrime == 1) {

&nbsp;       printf("%d is a prime number.\\n", num);

&nbsp;   } else {

&nbsp;       printf("%d is not a prime number.\\n", num);

&nbsp;   }



&nbsp;   return 0;

}
