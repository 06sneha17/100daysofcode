//Write a program to check if a number is a palindrome

\#include <stdio.h>



int main() {

&nbsp;   int number, reversed\_number = 0, original\_number, remainder;



&nbsp;   printf("Enter an integer: ");

&nbsp;   scanf("%d", \&number);



&nbsp;   original\_number = number; // Store the original number for comparison



&nbsp;   // Reverse the number

&nbsp;   while (number != 0) {

&nbsp;       remainder = number % 10; // Get the last digit

&nbsp;       reversed\_number = reversed\_number \* 10 + remainder; // Build the reversed number

&nbsp;       number /= 10; // Remove the last digit from the original number

&nbsp;   }



&nbsp;   // Compare the original and reversed numbers

&nbsp;   if (original\_number == reversed\_number) {

&nbsp;       printf("%d is a palindrome number.\\n", original\_number);

&nbsp;   } else {

&nbsp;       printf("%d is not a palindrome number.\\n", original\_number);

&nbsp;   }



&nbsp;   return 0;

}
