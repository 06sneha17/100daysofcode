//Write a program to print the following pattern

\#include <stdio.h>



int main() {

&nbsp;   int i, j, rows = 4; // 'rows' represents the number of rows in the upper half (or lower half)



&nbsp;   // Decreasing part of the hourglass

&nbsp;   for (i = rows; i >= 1; i--) {

&nbsp;       // Print leading spaces

&nbsp;       for (j = 1; j <= rows - i; j++) {

&nbsp;           printf(" ");

&nbsp;       }

&nbsp;       // Print asterisks

&nbsp;       for (j = 1; j <= 2 \* i - 1; j++) {

&nbsp;           printf("\*");

&nbsp;       }

&nbsp;       printf("\\n");

&nbsp;   }



&nbsp;   // Increasing part of the hourglass

&nbsp;   for (i = 2; i <= rows; i++) {

&nbsp;       // Print leading spaces

&nbsp;       for (j = 1; j <= rows - i; j++) {

&nbsp;           printf(" ");

&nbsp;       }

&nbsp;       // Print asterisks

&nbsp;       for (j = 1; j <= 2 \* i - 1; j++) {

&nbsp;           printf("\*");

&nbsp;       }

&nbsp;       printf("\\n");

&nbsp;   }



&nbsp;   return 0;

}
