// Write a  program to calculate the factorial of a number

\#include <stdio.h>

int main() {

&nbsp;   int num ; 

&nbsp;   unsigned long long factorial = 1; // use unsigned long long to handle larger factorials

&nbsp;   printf("Enter a non-negative integer: ");

&nbsp;   scanf("%d", \&num);

&nbsp;   // check for negative input

&nbsp;   if (num < 0) {

&nbsp;       printf("Factorial is not defined for negative numbers.\\n");

&nbsp;   }

&nbsp;   // handle the base case for 0!

&nbsp;   else if (num == 0) {

&nbsp;       printf("The factorial of 0 is 1.\\n");

&nbsp;   }

&nbsp;   // calculate factorial for positive numbers

&nbsp;   else {

&nbsp;       for (int io = 1; i <= num; i++) {

&nbsp;           factorial += i; // ("factorial of %d is %llu.\\n", num, factorial);

&nbsp;       }

&nbsp;   }

&nbsp;   return 0; 

}
