//Convert a lowercase string to uppercase without using built-in functions

\#include <stdio.h>



int main() {

&nbsp;   char str\[100]; // Declare a character array to store the string

&nbsp;   int i = 0;     // Loop counter



&nbsp;   printf("Enter a lowercase string: ");

&nbsp;   // Using fgets for safer input than gets

&nbsp;   fgets(str, sizeof(str), stdin); 



&nbsp;   // Iterate through the string until the null terminator is encountered

&nbsp;   while (str\[i] != '\\0') {

&nbsp;       // Check if the current character is a lowercase letter

&nbsp;       if (str\[i] >= 'a' \&\& str\[i] <= 'z') {

&nbsp;           // Convert lowercase to uppercase by subtracting 32 from its ASCII value

&nbsp;           str\[i] = str\[i] - 32;

&nbsp;       }

&nbsp;       i++; // Move to the next character

&nbsp;   }



&nbsp;   printf("Uppercase string: %s", str);



&nbsp;   return 0;

}
