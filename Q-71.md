

//Read and print a matrix

\#include <stdio.h>



int main() {

&nbsp;   int rows, cols;



&nbsp;   // Get the number of rows and columns from the user

&nbsp;   printf("Enter the number of rows: ");

&nbsp;   scanf("%d", \&rows);

&nbsp;   printf("Enter the number of columns: ");

&nbsp;   scanf("%d", \&cols);



&nbsp;   // Declare a 2D array (matrix) with the given dimensions

&nbsp;   int matrix\[rows]\[cols];



&nbsp;   // Read elements into the matrix

&nbsp;   printf("Enter the elements of the matrix:\\n");

&nbsp;   for (int i = 0; i < rows; i++) {

&nbsp;       for (int j = 0; j < cols; j++) {

&nbsp;           printf("Enter element at position \[%d]\[%d]: ", i, j);

&nbsp;           scanf("%d", \&matrix\[i]\[j]);

&nbsp;       }

&nbsp;   }



&nbsp;   // Print the matrix

&nbsp;   printf("\\nThe entered matrix is:\\n");

&nbsp;   for (int i = 0; i < rows; i++) {

&nbsp;       for (int j = 0; j < cols; j++) {

&nbsp;           printf("%d\\t", matrix\[i]\[j]); // Print element followed by a tab

&nbsp;       }

&nbsp;       printf("\\n"); // Move to the next line after printing each row

&nbsp;   }



&nbsp;   return 0;

}
