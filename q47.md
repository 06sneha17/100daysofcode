//Write a program to print the following pattern

\#include <stdio.h>



int main() {

&nbsp;   int rows, i, j;



&nbsp;   // Prompt the user to enter the number of rows

&nbsp;   printf("Enter the number of rows for the right-angled triangle: ");

&nbsp;   scanf("%d", \&rows);



&nbsp;   // Outer loop for managing rows

&nbsp;   for (i = 1; i <= rows; i++) {

&nbsp;       // Inner loop for printing stars in each row

&nbsp;       // The number of stars in each row is equal to the current row number (i)

&nbsp;       for (j = 1; j <= i; j++) {

&nbsp;           printf("\* ");

&nbsp;       }

&nbsp;       // Move to the next line after printing stars for the current row

&nbsp;       printf("\\n");

&nbsp;   }



&nbsp;   return 0;

}
