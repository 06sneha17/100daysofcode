//Search for an element in an array using linear search

\#include <stdio.h>



// Function to perform linear search

int linearSearch(int arr\[], int size, int target) {

&nbsp;   for (int i = 0; i < size; i++) {

&nbsp;       if (arr\[i] == target) {

&nbsp;           return i; // Return the index if target is found

&nbsp;       }

&nbsp;   }

&nbsp;   return -1; // Return -1 if target is not found

}



int main() {

&nbsp;   int numbers\[] = {10, 25, 5, 40, 15, 30};

&nbsp;   int size = sizeof(numbers) / sizeof(numbers\[0]);

&nbsp;   int searchElement = 15;



&nbsp;   int result = linearSearch(numbers, size, searchElement);



&nbsp;   if (result != -1) {

&nbsp;       printf("Element %d found at index %d.\\n", searchElement, result);

&nbsp;   } else {

&nbsp;       printf("Element %d not found in the array.\\n", searchElement);

&nbsp;   }



&nbsp;   searchElement = 50; // Searching for an element not in the array

&nbsp;   result = linearSearch(numbers, size, searchElement);



&nbsp;   if (result != -1) {

&nbsp;       printf("Element %d found at index %d.\\n", searchElement, result);

&nbsp;   } else {

&nbsp;       printf("Element %d not found in the array.\\n", searchElement);

&nbsp;   }



&nbsp;   return 0;

}
