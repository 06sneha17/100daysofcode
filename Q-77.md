//Check if the elements on the diagonal of a matrix are distinct

\#include <stdbool.h>



bool areDiagonalElementsDistinct(int matrix\[]\[N], int N) {

&nbsp;   for (int i = 0; i < N; i++) {

&nbsp;       for (int j = i + 1; j < N; j++) {

&nbsp;           if (matrix\[i]\[i] == matrix\[j]\[j]) {

&nbsp;               return false; // Found a duplicate

&nbsp;           }

&nbsp;       }

&nbsp;   }

&nbsp;   return true; // All diagonal elements are distinct

}
