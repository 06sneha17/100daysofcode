//Write a program to find the LCM of two numbers

\#include <stdio.h>



int main() {

&nbsp;   int num1, num2, max, lcm = 1;



&nbsp;   // Prompt user to enter two numbers

&nbsp;   printf("Enter two positive integers: ");

&nbsp;   scanf("%d %d", \&num1, \&num2);



&nbsp;   // Find the larger of the two numbers

&nbsp;   max = (num1 > num2) ? num1 : num2;



&nbsp;   // Start checking from 'max' to find the LCM

&nbsp;   // The loop continues indefinitely until an LCM is found

&nbsp;   while (1) {

&nbsp;       // Check if 'max' is divisible by both num1 and num2

&nbsp;       if (max % num1 == 0 \&\& max % num2 == 0) {

&nbsp;           lcm = max; // If divisible, 'max' is the LCM

&nbsp;           break;     // Exit the loop

&nbsp;       }

&nbsp;       // If not divisible, increment 'max' by itself to check the next possible multiple

&nbsp;       max += max; 

&nbsp;   }



&nbsp;   // Print the calculated LCM

&nbsp;   printf("The LCM of %d and %d is %d\\n", num1, num2, lcm);



&nbsp;   return 0;

}
