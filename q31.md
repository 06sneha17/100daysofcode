//Write a program to take a number as input and print its equivalent binary representation

\#include <stdio.h>



int main() {

&nbsp;   int decimalNumber;

&nbsp;   int binaryArray\[32]; // Array to store binary digits (sufficient for a 32-bit integer)

&nbsp;   int i = 0;



&nbsp;   printf("Enter a decimal number: ");

&nbsp;   scanf("%d", \&decimalNumber);



&nbsp;   // Handle the special case of 0

&nbsp;   if (decimalNumber == 0) {

&nbsp;       printf("Binary equivalent: 0\\n");

&nbsp;       return 0;

&nbsp;   }



&nbsp;   // Convert decimal to binary using repeated division by 2

&nbsp;   int tempNumber = decimalNumber;

&nbsp;   while (tempNumber > 0) {

&nbsp;       binaryArray\[i] = tempNumber % 2; // Store the remainder

&nbsp;       tempNumber /= 2;                // Divide by 2

&nbsp;       i++;

&nbsp;   }



&nbsp;   printf("Binary equivalent: ");

&nbsp;   // Print the binary digits in reverse order

&nbsp;   for (int j = i - 1; j >= 0; j--) {

&nbsp;       printf("%d", binaryArray\[j]);

&nbsp;   }

&nbsp;   printf("\\n");



&nbsp;   return 0;

}
