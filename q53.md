//Write a program to print the following pattern

\#include <stdio.h>



int main() {

&nbsp;   int n, i, j;

&nbsp;   int blockSize = 1;   // first block has size 1

&nbsp;   int line = 1;        // track line number

&nbsp;   int toggle = 1;      // 1 = print stars, 0 = print blanks



&nbsp;   printf("Enter number of lines: ");

&nbsp;   scanf("%d", \&n);



&nbsp;   while (line <= n) {

&nbsp;       for (i = 1; i <= blockSize \&\& line <= n; i++, line++) {

&nbsp;           if (toggle == 1)

&nbsp;               printf("\*\\n");

&nbsp;           else

&nbsp;               printf("\\n");

&nbsp;       }

&nbsp;       blockSize += 2;    // increase block size by odd number

&nbsp;       toggle = 1 - toggle; // switch between stars and blanks

&nbsp;   }



&nbsp;   return 0;

}
