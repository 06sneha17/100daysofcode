//Write a program to check if a number is a perfect number

\#include <stdio.h>



int main() {

&nbsp;   int number, i, sum = 0;



&nbsp;   // Prompt the user to enter a number

&nbsp;   printf("Enter a positive integer: ");

&nbsp;   scanf("%d", \&number);



&nbsp;   // Check if the number is positive

&nbsp;   if (number <= 0) {

&nbsp;       printf("Please enter a positive integer.\\n");

&nbsp;       return 1; // Indicate an error

&nbsp;   }



&nbsp;   // Calculate the sum of proper divisors

&nbsp;   for (i = 1; i < number; i++) {

&nbsp;       if (number % i == 0) {

&nbsp;           sum += i;

&nbsp;       }

&nbsp;   }



&nbsp;   // Check if the sum of divisors equals the number

&nbsp;   if (sum == number) {

&nbsp;       printf("%d is a perfect number.\\n", number);

&nbsp;   } else {

&nbsp;       printf("%d is not a perfect number.\\n", number);

&nbsp;   }



&nbsp;   return 0; // Indicate successful program execution

}
