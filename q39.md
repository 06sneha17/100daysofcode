//Write a program to find the product of odd digits of a number

\#include <stdio.h>



int main() {

&nbsp;   int num, digit;

&nbsp;   long long product = 1; // Use long long to handle potential larger products



&nbsp;   printf("Enter an integer: ");

&nbsp;   scanf("%d", \&num);



&nbsp;   if (num == 0) { // Special case for 0

&nbsp;       printf("Product of odd digits: 0\\n");

&nbsp;       return 0;

&nbsp;   }

&nbsp;   

&nbsp;   // Handle negative numbers by converting to positive for digit extraction

&nbsp;   if (num < 0) {

&nbsp;       num = -num;

&nbsp;   }



&nbsp;   int hasOddDigit = 0; // Flag to check if any odd digit was found



&nbsp;   while (num > 0) {

&nbsp;       digit = num % 10; // Get the last digit

&nbsp;       if (digit % 2 != 0) { // Check if the digit is odd

&nbsp;           product \*= digit; // Multiply if it's odd

&nbsp;           hasOddDigit = 1; // Set flag to true

&nbsp;       }

&nbsp;       num /= 10; // Remove the last digit

&nbsp;   }



&nbsp;   if (hasOddDigit) {

&nbsp;       printf("Product of odd digits: %lld\\n", product);

&nbsp;   } else {

&nbsp;       printf("No odd digits found in the number.\\n");

&nbsp;   }



&nbsp;   return 0;

}
