//Find the digit that occurs the most times in an integer number

\#include <stdio.h>

\#include <limits.h> // For INT\_MIN



int main() {

&nbsp;   int number;

&nbsp;   printf("Enter an integer: ");

&nbsp;   scanf("%d", \&number);



&nbsp;   // Handle negative numbers by taking the absolute value

&nbsp;   if (number < 0) {

&nbsp;       number = -number;

&nbsp;   }



&nbsp;   // Handle the case of 0 separately, as the loop won't run for it

&nbsp;   if (number == 0) {

&nbsp;       printf("The most frequent digit is: 0\\n");

&nbsp;       return 0;

&nbsp;   }



&nbsp;   int digit\_counts\[10] = {0}; // Initialize all counts to 0



&nbsp;   // Extract digits and count their occurrences

&nbsp;   while (number > 0) {

&nbsp;       int digit = number % 10;

&nbsp;       digit\_counts\[digit]++;

&nbsp;       number /= 10;

&nbsp;   }



&nbsp;   int max\_frequency = INT\_MIN;

&nbsp;   int most\_frequent\_digit = -1;



&nbsp;   // Find the digit with the maximum frequency

&nbsp;   for (int i = 0; i < 10; i++) {

&nbsp;       if (digit\_counts\[i] > max\_frequency) {

&nbsp;           max\_frequency = digit\_counts\[i];

&nbsp;           most\_frequent\_digit = i;

&nbsp;       }

&nbsp;   }



&nbsp;   printf("The most frequent digit is: %d\\n", most\_frequent\_digit);



&nbsp;   return 0;

}
