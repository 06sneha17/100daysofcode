//count even and odd numbers in an array in c 

\#include <stdio.h>



int main() {

&nbsp;   int n; // Size of the array

&nbsp;   printf("Enter the size of the array: ");

&nbsp;   scanf("%d", \&n);



&nbsp;   int arr\[n]; // Declare the array

&nbsp;   printf("Enter the array elements:\\n");

&nbsp;   for (int i = 0; i < n; i++) {

&nbsp;       scanf("%d", \&arr\[i]);

&nbsp;   }



&nbsp;   int even\_count = 0; // Counter for even numbers

&nbsp;   int odd\_count = 0;  // Counter for odd numbers



&nbsp;   for (int i = 0; i < n; i++) {

&nbsp;       if (arr\[i] % 2 == 0) { // Check if the number is even

&nbsp;           even\_count++;

&nbsp;       } else { // If not even, it's odd

&nbsp;           odd\_count++;

&nbsp;       }

&nbsp;   }



&nbsp;   printf("Number of even elements: %d\\n", even\_count);

&nbsp;   printf("Number of odd elements: %d\\n", odd\_count);



&nbsp;   return 0;

}
