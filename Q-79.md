//Perform diagonal traversal of a matrix

\#include <stdio.h>



void printDiagonalTraversal(int matrix\[]\[3], int rows, int cols) {

&nbsp;   // Iterate through all possible sums of i + j

&nbsp;   for (int sum = 0; sum <= rows + cols - 2; sum++) {

&nbsp;       // For each sum, iterate through rows

&nbsp;       for (int i = 0; i < rows; i++) {

&nbsp;           int j = sum - i; // Calculate corresponding column index



&nbsp;           // Check if j is within bounds

&nbsp;           if (j >= 0 \&\& j < cols) {

&nbsp;               printf("%d ", matrix\[i]\[j]);

&nbsp;           }

&nbsp;       }

&nbsp;   }

&nbsp;   printf("\\n");

}



int main() {

&nbsp;   int matrix\[3]\[3] = {

&nbsp;       {1, 2, 3},

&nbsp;       {7, 8, 9}

&nbsp;   };

&nbsp;   int rows = 3;

&nbsp;   int cols = 3;



&nbsp;   printf("Diagonal Traversal (main diagonals): ");
