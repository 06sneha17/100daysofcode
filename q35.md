//Write a program to print all factors of a given number 

\#include <stdio.h>



int main() {

&nbsp;   int num, i;



&nbsp;   // Prompt the user to enter a positive integer

&nbsp;   printf("Enter a positive integer: ");

&nbsp;   scanf("%d", \&num);



&nbsp;   // Print a message indicating the factors

&nbsp;   printf("Factors of %d are: ", num);



&nbsp;   // Loop from 1 up to the number itself

&nbsp;   for (i = 1; i <= num; ++i) {

&nbsp;       // Check if 'i' divides 'num' without a remainder

&nbsp;       if (num % i == 0) {

&nbsp;           // If 'i' is a factor, print it

&nbsp;           printf("%d ", i);

&nbsp;       }

&nbsp;   }



&nbsp;   printf("\\n"); // Print a newline for better formatting

&nbsp;   return 0;

}
