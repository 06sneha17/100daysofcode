//Count positive, negative, and zero elements in an array

\#include <stdio.h>



int main() {

&nbsp;   int arr\[] = {2, -1, 5, 0, -3, 8, 0, -7};

&nbsp;   int n = sizeof(arr) / sizeof(arr\[0]); // Calculate array size



&nbsp;   int positive\_count = 0;

&nbsp;   int negative\_count = 0;

&nbsp;   int zero\_count = 0;



&nbsp;   for (int i = 0; i < n; i++) {

&nbsp;       if (arr\[i] > 0) {

&nbsp;           positive\_count++;

&nbsp;       } else if (arr\[i] < 0) {

&nbsp;           negative\_count++;

&nbsp;       } else { // arr\[i] == 0

&nbsp;           zero\_count++;

&nbsp;       }

&nbsp;   }



&nbsp;   printf("Number of positive elements: %d\\n", positive\_count);

&nbsp;   printf("Number of negative elements: %d\\n", negative\_count);

&nbsp;   printf("Number of zero elements: %d\\n", zero\_count);



&nbsp;   return 0;

}
