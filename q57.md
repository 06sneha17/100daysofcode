// Function to calculate the sum of array elements

int sumArrayElements(int arr\[], int n) {

&nbsp;   int sum = 0; // Initialize sum to 0

&nbsp;   for (int i = 0; i < n; i++) {

&nbsp;       sum += arr\[i]; // Add each element to the sum

&nbsp;   }

&nbsp;   return sum; // Return the calculated sum

}
