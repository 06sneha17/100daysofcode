

//Rotate an array to the right by k positions

\#include <stdio.h>

\#include <stdlib.h> 



void rotateRight(int arr\[], int n, int k) {

&nbsp;   if (n == 0 || k == 0) {

&nbsp;       return; // No rotation needed for empty array or k=0

&nbsp;   }



&nbsp;   k = k % n; // Handle k greater than or equal to n



&nbsp;   if (k == 0) {

&nbsp;       return; // No effective rotation after modulo

&nbsp;   }



&nbsp;   // Allocate memory for a temporary array to store the last k elements

&nbsp;   int \*temp = (int \*)malloc(k \* sizeof(int));

&nbsp;   if (temp == NULL) {

&nbsp;       printf("Memory allocation failed!\\n");

&nbsp;       return;

&nbsp;   }



&nbsp;   // Copy the last k elements to the temporary array

&nbsp;   for (int i = 0; i < k; i++) {

&nbsp;       temp\[i] = arr\[n - k + i];

&nbsp;   }



&nbsp;   // Shift the remaining n-k elements to the right

&nbsp;   for (int i = n - k - 1; i >= 0; i--) {

&nbsp;       arr\[i + k] = arr\[i];

&nbsp;   }



&nbsp;   // Copy the elements from the temporary array to the beginning of the original array

&nbsp;   for (int i = 0; i < k; i++) {

&nbsp;       arr\[i] = temp\[i];

&nbsp;   }



&nbsp;   free(temp); // Free the allocated memory

}



int main() {

&nbsp;   int arr\[] = {1, 2, 3, 4, 5, 6, 7};

&nbsp;   int n = sizeof(arr) / sizeof(arr\[0]);

&nbsp;   int k = 2; // Rotate by 2 positions to the right



&nbsp;   printf("Original array: ");

&nbsp;   for (int i = 0; i < n; i++) {

&nbsp;       printf("%d ", arr\[i]);

&nbsp;   }

&nbsp;   printf("\\n");



&nbsp;   rotateRight(arr, n, k);



&nbsp;   printf("Array after rotating right by %d positions: ", k);

&nbsp;   for (int i = 0; i < n; i++) {

&nbsp;       printf("%d ", arr\[i]);

&nbsp;   }

&nbsp;   printf("\\n");



&nbsp;   return 0;

}
