//Write a program to find the sum of the series: 1 + 3/4 + 5/6 + 7/8 + â€¦ up to n terms

\#include <stdio.h>



int main() {

&nbsp;   int n, i;

&nbsp;   double sum = 0.0;

&nbsp;   int numerator, denominator;



&nbsp;   printf("Enter the number of terms (n): ");

&nbsp;   scanf("%d", \&n);



&nbsp;   for (i = 1; i <= n; i++) {

&nbsp;       numerator = 2 \* i - 1;

&nbsp;       denominator = 2 \* i;

&nbsp;       sum += (double)numerator / denominator;

&nbsp;   }



&nbsp;   printf("The sum of the series up to %d terms is: %.6lf\\n", n, sum);



&nbsp;   return 0;

}
