//Write a program to find the sum of digits of a number

\#include <stdio.h>



int main() {

&nbsp;   int number, sum = 0, digit;



&nbsp;   // Prompt the user to enter a number

&nbsp;   printf("Enter a number: ");

&nbsp;   scanf("%d", \&number);



&nbsp;   // Loop to extract digits and calculate their sum

&nbsp;   while (number != 0) {

&nbsp;       digit = number % 10; // Extract the last digit

&nbsp;       sum += digit;        // Add the digit to the sum

&nbsp;       number /= 10;        // Remove the last digit from the number

&nbsp;   }



&nbsp;   // Display the sum of digits

&nbsp;   printf("Sum of the digits: %d\\n", sum);



&nbsp;   return 0;

}
