\#include <stdio.h>

\#include <string.h>



int main() {

&nbsp;   char binaryNumber\[100]; // To store the input binary number

&nbsp;   char onesComplement\[100]; // To store the calculated 1's complement

&nbsp;   int i, length;



&nbsp;   printf("Enter a binary number: ");

&nbsp;   scanf("%s", binaryNumber);



&nbsp;   length = strlen(binaryNumber);



&nbsp;   // Iterate through the binary number and find its 1's complement

&nbsp;   for (i = 0; i < length; i++) {

&nbsp;       if (binaryNumber\[i] == '0') {

&nbsp;           onesComplement\[i] = '1';

&nbsp;       } else if (binaryNumber\[i] == '1') {

&nbsp;           onesComplement\[i] = '0';

&nbsp;       } else {

&nbsp;           printf("Error: Invalid binary digit entered.\\n");

&nbsp;           return 1; // Exit with an error code

&nbsp;       }

&nbsp;   }



&nbsp;   onesComplement\[length] = '\\0'; // Null-terminate the complement string



&nbsp;   printf("Original binary number: %s\\n", binaryNumber);

&nbsp;   printf("1's complement: %s\\n", onesComplement);



&nbsp;   return 0; // Indicate successful execution

}
