//Read and print elements of a one-dimensional array

\#include <stdio.h>



int main() {

&nbsp;   int n, i;



&nbsp;   // Prompt the user for the number of elements

&nbsp;   printf("Enter the number of elements for the array: ");

&nbsp;   scanf("%d", \&n);



&nbsp;   // Declare an array of size n

&nbsp;   int arr\[n];



&nbsp;   // Input elements into the array

&nbsp;   printf("Enter %d integer elements:\\n", n);

&nbsp;   for (i = 0; i < n; i++) {

&nbsp;       printf("Element %d: ", i + 1);

&nbsp;       scanf("%d", \&arr\[i]);

&nbsp;   }



&nbsp;   // Print the elements of the array

&nbsp;   printf("\\nThe elements in the array are: ");

&nbsp;   for (i = 0; i < n; i++) {

&nbsp;       printf("%d ", arr\[i]);

&nbsp;   }

&nbsp;   printf("\\n");



&nbsp;   return 0;

}
