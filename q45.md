//Write a program to find the sum of the series: 2/3 + 4/7 + 6/11 + 8/15 + ... up to n terms

\#include <stdio.h>



int main() {

&nbsp;   int n;

&nbsp;   double sum = 0.0;

&nbsp;   int numerator = 2;

&nbsp;   int denominator = 3;



&nbsp;   printf("Enter the number of terms (n): ");

&nbsp;   scanf("%d", \&n);



&nbsp;   for (int i = 1; i <= n; i++) {

&nbsp;       sum += (double)numerator / denominator;

&nbsp;       numerator += 2;

&nbsp;       denominator += 4;

&nbsp;   }



&nbsp;   printf("The sum of the series up to %d terms is: %.4lf\\n", n, sum);



&nbsp;   return 0;

}
