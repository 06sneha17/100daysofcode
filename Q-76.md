//Check if a matrix is symmetric

\#include <stdio.h>

\#include <stdbool.h> 



// Function to check if a matrix is symmetric

bool isSymmetric(int matrix\[]\[10], int rows, int cols) {

&nbsp;   // A symmetric matrix must be a square matrix

&nbsp;   if (rows != cols) {

&nbsp;       return false;

&nbsp;   }



&nbsp;   // Compare elements with their transposed counterparts

&nbsp;   for (int i = 0; i < rows; i++) {

&nbsp;       for (int j = 0; j < cols; j++) {

&nbsp;           if (matrix\[i]\[j] != matrix\[j]\[i]) {

&nbsp;               return false; // Found a non-matching pair, so it's not symmetric

&nbsp;           }

&nbsp;       }

&nbsp;   }



&nbsp;   return true; // All elements matched, so it's symmetric

}



int main() {

&nbsp;   int rows, cols;



&nbsp;   printf("Enter the number of rows: ");

&nbsp;   scanf("%d", \&rows);

&nbsp;   printf("Enter the number of columns: ");

&nbsp;   scanf("%d", \&cols);



&nbsp;   int matrix\[10]\[10]; // Assuming max size 10x10 for simplicity



&nbsp;   printf("Enter the elements of the matrix:\\n");

&nbsp;   for (int i = 0; i < rows; i++) {

&nbsp;       for (int j = 0; j < cols; j++) {

&nbsp;           scanf("%d", \&matrix\[i]\[j]);

&nbsp;       }

&nbsp;   }



&nbsp;   if (isSymmetric(matrix, rows, cols)) {

&nbsp;       printf("The matrix is symmetric.\\n");

&nbsp;   } else {

&nbsp;       printf("The matrix is not symmetric.\\n");

&nbsp;   }



&nbsp;   return 0;

}
