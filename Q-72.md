

//Find the sum of all elements in a matrix

\#include <stdio.h>



int sumArrayElements(int arr\[], int n) {

&nbsp;   int sum = 0; // Initialize sum to 0

&nbsp;   for (int i = 0; i < n; i++) {

&nbsp;       sum += arr\[i]; // Add each element to sum

&nbsp;   }

&nbsp;   return sum; // Return the calculated sum

}



int main() {

&nbsp;   int arr\[] = {10, 20, 30, 40, 50}; // Example array

&nbsp;   int n = sizeof(arr) / sizeof(arr\[0]); // Calculate the number of elements



&nbsp;   int totalSum = sumArrayElements(arr, n); // Call the function to get the sum



&nbsp;   printf("The sum of array elements is: %d\\n", totalSum); // Print the result



&nbsp;   return 0;

}
