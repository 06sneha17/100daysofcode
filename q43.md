//Write a program to check if a number is a strong number

\#include <stdio.h>



// Function to calculate factorial of a number

long long factorial(int n) {

&nbsp;   long long fact = 1;

&nbsp;   for (int i = 1; i <= n; i++) {

&nbsp;       fact \*= i;

&nbsp;   }

&nbsp;   return fact;

}



// Function to check if a number is strong

int isStrong(int num) {

&nbsp;   int originalNum = num;

&nbsp;   long long sumOfFactorials = 0;

&nbsp;   int digit;



&nbsp;   while (num > 0) {

&nbsp;       digit = num % 10; // Extract the last digit

&nbsp;       sumOfFactorials += factorial(digit); // Add factorial of the digit to the sum

&nbsp;       num /= 10; // Remove the last digit

&nbsp;   }



&nbsp;   if (sumOfFactorials == originalNum) {

&nbsp;       return 1; // It's a strong number

&nbsp;   } else {

&nbsp;       return 0; // It's not a strong number

&nbsp;   }

}



int main() {

&nbsp;   int number;



&nbsp;   printf("Enter a number: ");

&nbsp;   scanf("%d", \&number);



&nbsp;   if (isStrong(number)) {

&nbsp;       printf("%d is a strong number.\\n", number);

&nbsp;   } else {

&nbsp;       printf("%d is not a strong number.\\n", number);

&nbsp;   }



&nbsp;   return 0;
