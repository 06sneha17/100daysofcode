// Write a  program to print the sum of the first n odd numbers

\#include <stdio.h>

int main() {

&nbsp;   int n; // variable to store the number of odd terms

&nbsp;   int sum = 0; // variable to store the sum of odd numbers

&nbsp;   int i; // loop counter

&nbsp;   // prompt the user to enter the value of n

&nbsp;   printf("Enter the number of odd terms (n) : ");

&nbsp;   scanf("%d", \&n);

&nbsp;   // validate input: ensure n is a positive integer

&nbsp;   if (n <= 0) {

&nbsp;       printf("Please enter a positive integer for n.\\n");

&nbsp;       return 1; // indicate an error

&nbsp;   }

&nbsp;   // calculate the sum of the first n odd numbers

&nbsp;   // the k-th odd number is given by 2\*k - 1

&nbsp;   for(i = 1; i <= n; i++) {

&nbsp;       sum += (2 \* i - 1);

&nbsp;   }

&nbsp; }

&nbsp; // print the sum

printf("The sum of the first %d odd number is: %d\\n" , n, sum) ;

&nbsp;   return 0; // indicate successful execution

}
