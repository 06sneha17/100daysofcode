

//Find the sum of each row of a matrix and store it in an array

\#include <stdio.h>



int main() {

&nbsp;   int rows, cols;



&nbsp;   // Get matrix dimensions from the user

&nbsp;   printf("Enter the number of rows: ");

&nbsp;   scanf("%d", \&rows);

&nbsp;   printf("Enter the number of columns: ");

&nbsp;   scanf("%d", \&cols);



&nbsp;   int matrix\[rows]\[cols];

&nbsp;   int row\_sums\[rows]; // Array to store sum of each row



&nbsp;   // Input matrix elements

&nbsp;   printf("Enter the elements of the matrix:\\n");

&nbsp;   for (int i = 0; i < rows; i++) {

&nbsp;       for (int j = 0; j < cols; j++) {

&nbsp;           printf("Enter element at matrix\[%d]\[%d]: ", i, j);

&nbsp;           scanf("%d", \&matrix\[i]\[j]);

&nbsp;       }

&nbsp;   }



&nbsp;   // Calculate sum of each row and store in row\_sums array

&nbsp;   for (int i = 0; i < rows; i++) {

&nbsp;       int current\_row\_sum = 0;

&nbsp;       for (int j = 0; j < cols; j++) {

&nbsp;           current\_row\_sum += matrix\[i]\[j];

&nbsp;       }

&nbsp;       row\_sums\[i] = current\_row\_sum;

&nbsp;   }



&nbsp;   // Print the row sums

&nbsp;   printf("\\nSum of each row:\\n");

&nbsp;   for (int i = 0; i < rows; i++) {

&nbsp;       printf("Row %d sum: %d\\n", i + 1, row\_sums\[i]);

&nbsp;   }



&nbsp;   return 0;

}
