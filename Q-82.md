//Print each character of a string on a new line

\#include <stdio.h>



int main() {

&nbsp;   char str\[100]; // Declare a character array to store the string

&nbsp;   int i = 0;     // Initialize a counter variable



&nbsp;   printf("Enter a string: ");

&nbsp;   scanf("%s", str); // Read the string from the user



&nbsp;   printf("Individual characters on new lines:\\n");



&nbsp;   // Loop through the string until the null terminator ('\\0') is encountered

&nbsp;   while (str\[i] != '\\0') {

&nbsp;       printf("%c\\n", str\[i]); // Print the current character followed by a newline

&nbsp;       i++;                    // Increment the counter

&nbsp;   }



&nbsp;   return 0;

&nbsp; }
