//Add two matrices

\#include <stdio.h> 



int main() {

&nbsp;   int row, col, i, j; // Variables for rows, columns, and loop counters

&nbsp;   int matrix1\[100]\[100], matrix2\[100]\[100], sum\[100]\[100]; // Declare 2D arrays for the matrices



&nbsp;   // Get the number of rows from the user

&nbsp;   printf("Enter the number of rows (1-100): ");

&nbsp;   scanf("%d", \&row);



&nbsp;   // Get the number of columns from the user

&nbsp;   printf("Enter the number of columns (1-100): ");

&nbsp;   scanf("%d", \&col);



&nbsp;   printf("\\nEnter elements of Matrix 1:\\n");

&nbsp;   // Input elements for the first matrix

&nbsp;   for (i = 0; i < row; ++i) {

&nbsp;       for (j = 0; j < col; ++j) {

&nbsp;           printf("Enter element a%d%d: ", i + 1, j + 1);

&nbsp;           scanf("%d", \&matrix1\[i]\[j]);

&nbsp;       }

&nbsp;   }



&nbsp;   printf("\\nEnter elements of Matrix 2:\\n");

&nbsp;   // Input elements for the second matrix

&nbsp;   for (i = 0; i < row; ++i) {

&nbsp;       for (j = 0; j < col; ++j) {

&nbsp;           printf("Enter element b%d%d: ", i + 1, j + 1);

&nbsp;           scanf("%d", \&matrix2\[i]\[j]);

&nbsp;       }

&nbsp;   }



&nbsp;   // Add the two matrices element by element

&nbsp;   for (i = 0; i < row; ++i) {

&nbsp;       for (j = 0; j < col; ++j) {

&nbsp;           sum\[i]\[j] = matrix1\[i]\[j] + matrix2\[i]\[j];

&nbsp;       }

&nbsp;   }



&nbsp;   // Print the resulting sum matrix

&nbsp;   printf("\\nSum of the two matrices: \\n");

&nbsp;   for (i = 0; i < row; ++i) {

&nbsp;       for (j = 0; j < col; ++j) {

&nbsp;           printf("%d ", sum\[i]\[j]);

&nbsp;           // Print a new line after each row for better formatting

&nbsp;           if (j == col - 1) {

&nbsp;               printf("\\n");

&nbsp;           }

&nbsp;       }

&nbsp;   }



&nbsp;   return 0; // Indicate successful execution

}
