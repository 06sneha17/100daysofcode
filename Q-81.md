//Count characters in a string without using built-in length functions

\#include <stdio.h>



int main() {

&nbsp;   char str\[100]; // Declare a character array to store the string

&nbsp;   int count = 0; // Initialize a counter for the characters

&nbsp;   int i = 0;     // Initialize a loop counter



&nbsp;   printf("Enter a string: ");

&nbsp;   scanf("%s", str); // Read the string from the user



&nbsp;   // Loop through the string until the null terminator ('\\0') is encountered

&nbsp;   while (str\[i] != '\\0') {

&nbsp;       count++; // Increment the character count

&nbsp;       i++;     // Move to the next character

&nbsp;   }



&nbsp;   printf("Number of characters in the string: %d\\n", count);



&nbsp;   return 0;

}
