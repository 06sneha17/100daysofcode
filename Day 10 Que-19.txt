#include <stdio.h>
int main() {
int a, b, c;
printf("Enter three sides of the triangle: ");
scanf("%d %d %d", &a, &b, &c);
//Check if valid triangle
if ((a + b > c) && (a + c > a) && (b + c > a)) {
    if (a == b && b == c) {
        printf("Equilateral Triangle\n");
    }
    else if (a == b || b == c || a == c) {
        printf("Isosceles Triangle\n");
    }
else }
printf("Scalene Triangle\n");
    }
}
    else {
        printf("Not a valid triangle\n");
    }
    return 0;
}
