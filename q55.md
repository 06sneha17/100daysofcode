//Write a program to print all the prime numbers from 1 to n.

\#include <stdio.h>



int main() {

&nbsp;   int n, i, j, is\_prime;



&nbsp;   // Prompt the user to enter the upper limit

&nbsp;   printf("Enter a positive integer (n): ");

&nbsp;   scanf("%d", \&n);



&nbsp;   printf("Prime numbers from 1 to %d are: ", n);



&nbsp;   // Iterate from 2 to n to check each number for primality

&nbsp;   for (i = 2; i <= n; i++) {

&nbsp;       is\_prime = 1; // Assume the current number is prime



&nbsp;       // Check for divisibility from 2 up to the square root of i

&nbsp;       // (Optimization: we only need to check divisors up to sqrt(i))

&nbsp;       for (j = 2; j \* j <= i; j++) { 

&nbsp;           if (i % j == 0) {

&nbsp;               is\_prime = 0; // If divisible, it's not prime

&nbsp;               break;        // Exit the inner loop as it's not prime

&nbsp;           }

&nbsp;       }



&nbsp;       // If is\_prime is still 1, the number is prime

&nbsp;       if (is\_prime == 1) {

&nbsp;           printf("%d ", i);

&nbsp;       }

&nbsp;   }



&nbsp;   printf("\\n"); // Print a newline at the end for better formatting



&nbsp;   return 0; // Indicate successful execution

}
